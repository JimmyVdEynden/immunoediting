Running the analysis on other mutation data

Van den Eynden Jimmy

02/04/2019

This document shortly explains how to 1) apply the HLA-binding annotation and 2) calculate HBMR and dNnonHLA/dNHLA ratios using your own mutation data. It will use the TCGA MC3 calls as an example but can be easily adapted to any other dataset.

Download example dataset

We will use the TCGA MC3 calls as reported by Ellrott et al. 2018 as an example dataset here. These data are publicly available at *https://gdc.cancer.gov/about-data/publications/mc3-2017*

# Takes some time to download, +/- 1GB
wget https://api.gdc.cancer.gov/data/1c8cfe5f-e52d-41ba-94da-f15ea1337efc -O downloads/MC3.maf

Add HLA annotation to mutation data

# maf_GP<- reaRDS(file = "downloads/maf_MC3_GP.rds")

# Load GRanges files containing information on all annotated regions
load("../data/annotation_gr.RData")

# Add HLA-binding annotation to maf
seqlevels(maf_GP)<- paste0("chr",unique(as.character(seqnames(maf_GP)))) # Match seqlevels first if necessary 
maf_GP_present_idx<- findOverlaps(maf_GP,annotation_gr_all,ignore.strand=T)
maf_GP_HLA_idx<- findOverlaps(maf_GP,annotation_gr_HLA_Mean,ignore.strand=T)
maf_GP$isHA<- (1:length(maf_GP))%in%queryHits(maf_GP_HLA_idx)
maf_GP$isHA[!(1:length(maf_GP))%in%queryHits(maf_GP_present_idx)]<- NA # Remove CDS position with no information
cat("HLA-binding mutations: ", 100*signif(mean(maf_GP$isHA,na.rm=T),3),"%"," \n")

Calculate metrics

# Calculate for cancers studied in the manuscript.
cancers<- sort(c('BRCA','COADREAD','LUAD','OV','HNSC','UCEC','LUSC','PRAD','LGG','STAD','SKCM','GBM','BLCA','LIHC','CESC','KIRP','SARC','PAAD','ESCA'))

# HBMR metric
source("../scripts/functions/calculate_HBMR.R")
subst_type_matrix_ls<- readRDS(file="../data/subst_type_matrix_ls.rds") # Expected mut per subst type
HBMR<- calculate_HBMR(sample = maf_GP$Tumor_Sample_Barcode,
                      variant = maf_GP$Variant_Classification,
                      isHA = maf_GP$isHA,
                      subst_type3 = maf_GP$subst_type3,
                      exp_mut = subst_type_matrix_ls$triNT,
                      cancer =  maf_GP$Cancer)[cancers,]
HBMR<- HBMR[order(HBMR[,"HBMR_norm"]),] # Sort by HBMRnorm

# dNHLA/dNnonHLA metric (shortly named Rsim here)
source("../scripts/functions/calculate_Rsim2.R")
subst_type_matrix<- readRDS(file="../data/subst_type_matrix_ls.rds")$triNT
Rsim_ls<- calculate_Rsim2(sample = maf_GP$Tumor_Sample_Barcode,
                        variant = maf_GP$Variant_Classification,
                        isHA = maf_GP$isHA,
                        subst_type3 = maf_GP$subst_type3,
                        exp_mut = subst_type_matrix,
                        cancer = maf_GP$Cancer)
Rsim<- Rsim_ls[["Rsim"]]

# Save metrics
save(HBMR,Rsim,file="results/MC3_metrics.RData")

dNHLA/dNnonHLA metric

source("../scripts/functions/cancer_cols.R")
Rsim_med<- sort(tapply(Rsim,TCGA_cancer_id[names(Rsim),1],"median",na.rm=T)[cancers],decreasing = T)
boxplot(Rsim~factor(TCGA_cancer_id[names(Rsim),1],levels = names(Rsim_med)),beside = T,las=2,outline=F,horizontal=T,frame.plot=F,col=c25[names(Rsim_med)],staplewex=0,notch=T,xlab="dNHLA/dNnonHLA",main="dNHLA/dNnonHLA metric \n (prototypical genotype using wild-type protein affinities)",ylim=c(0,2))
Rsim_p<- tapply(Rsim,TCGA_cancer_id[names(Rsim),1],function(x) wilcox.test(x,mu = 1)$p.value)[names(Rsim_med)]
## Warning in wilcox.test.default(x, mu = 1): cannot compute exact p-value
## with ties
Rsim_q<- p.adjust(Rsim_p,"fdr")
if(sum(Rsim_q<0.1)!=0) text(rep(1.8,sum(Rsim_q<0.1)),which(Rsim_q<0.1),paste0("P=",signif(Rsim_p[which(Rsim_q<0.1)],0.3)),xpd=NA)
abline(v=1,lty=2)

Note about calculating dNHLA/dNnonHLA metric for genotype-specific affinities and/or mutated peptides

Note that these values can also be calculated for genotype-specific and/or mutated peptide HLA affinities. This requires HLA genotyping of samples, prediction of mutated peptide affinities and calculation of expected ratios between HLA-binding and non-binding regions for each sample. See main code for more information or contact the corresponding author.
